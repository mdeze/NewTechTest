$(document).ready(function() {

    // init table sorting
    $("#demoTable").tablesorter({
      widgets: ['zebra'],
      headers: {
            4: { sorter: false },
            5: {
              sorter:'month-year-date'
            },
            6: {
              sorter:'large-currency'
            },
            7: {
            sorter:'large-currency'
            },
            8: { sorter: false }
      }                
    });
});     

// 
// currency sorting support
//
$.tablesorter.addParser({
   id: 'large-currency',
   is: function(s) {
           return false;
   },
   format: function(s) {
       var new_currency = s.replace(new RegExp(/[£$€]/g), ""); 

       return $.tablesorter.formatFloat(new_currency.replace(/,/gi, ''));
   },
   type: 'numeric'
});

// 
// month/year sorting support
//
$.tablesorter.addParser({
   id: 'month-year-date',
   is: function(s) {
           return false;
   },
   format: function(s) {
       var dateSplit = s.split('/');

       if(2 !== dateSplit.length)
               return 0;

       return new Date(dateSplit[1], dateSplit[0], 1);
   },
   type: 'numeric'
});
