Bootstrap Framework Demo

Confirm Modal Dialog

Web Site A callback function is called when the confirm button is clicked

Example - Regular Usage

// load plugin
<script src="js/bootstrap-confirm.js"></script>

// init confirm dialog
$('#confirmBtn').click(function(){
  $('#confirmAction').confirmModal({
    heading: 'Confirm Your Action',
    body: 'Action confirmation details',
    callback: function(){
      console.log('you confirm dialog called the callback');
    }
  });
});

// 
<div id="confirmAction"></div>
